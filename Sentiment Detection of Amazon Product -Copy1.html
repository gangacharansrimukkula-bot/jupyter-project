# Function to assign sentiment (1 for Positive, 0 for Negative)
def assign_sentiment(rating):
    try:
        rating = float(rating)
        return 1 if rating >= 4 else 0
    except ValueError:
        return np.nan  # Handle invalid ratings

# Create sentiment dataframe
sdf = df[['rating', 'reviewText']].copy()
sdf['sentiment'] = sdf['rating'].apply(assign_sentiment)
sdf.drop('rating', axis=1, inplace=True)
sdf.dropna(subset=['sentiment'], inplace=True)  # Drop invalid sentiment rows

# Initialize stemmer and stopwords
ps = PorterStemmer()
stop_words = set(stopwords.words('english'))

# Corpus to hold cleaned reviews
corpus = []

# Process each review
for i in range(len(sdf)):
    review = re.sub('[^a-zA-Z]', ' ', str(sdf['reviewText'].iloc[i]))  # Handle non-string
    review = review.lower()
    review = review.split()
    review = [ps.stem(word) for word in review if word not in stop_words]
    review = ' '.join(review)
    corpus.append(review)

# Display first few cleaned reviews
print("Cleaned reviews sample:", corpus[:5])

def predict_sentiment(new_review, model, vectorizer):
    review = re.sub('[^a-zA-Z]', ' ', new_review)
    review = review.lower()
    review = review.split()
    review = [ps.stem(word) for word in review if word not in stop_words]
    review = ' '.join(review)
    
    X_new = vectorizer.transform([review]).toarray()
    pred = model.predict(X_new)[0]
    return 'Positive' if pred == 1 else 'Negative'
